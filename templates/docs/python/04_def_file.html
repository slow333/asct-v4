함수 사용 방법

def add_many(*args):
  result = 0
  for arg in args:
    result += arg
  return result

add_many(1,2,3,4,5,6,7)

def add_mul(choice, *args):
  if choice == "add":
    result = 0
    for arg in args:
      result += arg
  elif choice == "mul":
    result = 1
    for arg in args:
      result *= arg
  return result

add_mul("add", 1,2,3,4,5)
add_mul("mul", 1,2,3,4,5)

*args

나머지 매게변수로 동작함

**kwargs

입력값을 dictionary 형태로 처리

return a, b 하면 2개의 값을 tuple로 반환함

def print_kwargs(**kwargs):
  print(kwargs)

print_kwargs(a=1, b=2, c=3, d=4) 

def add_and_mul(a,b):
  return a+b, a*b

add_and_mul(3,4)
result1, result2 = add_and_mul(3,4)
print(result1, result2)

readlines() 함수 사용하기

한글 깨지 방지를 위해 쓰기에 encoding=‘utf-8’ 필요
읽을 때도 encoding=‘utf-8’ 필요

f = open("새파일.txt", "w", encoding="utf-8")
for i in range(1, 6):
    data = "%d번째 줄입니다.\n" % i
    f.write(data)
f.close()

f = open("새파일.txt", "r", encoding="utf-8")
lines = f.readlines()
print(lines[len(lines) - 1])  # Print the last line
for line in lines:
    print(line)
f.close()

with문과 함께 사용하기(close()를 안해도 됨)

with open("external.txt", "w") as f:
  f.write("Hello, World!")
with open("external.txt", "r") as f:
  content = f.read()
  print(content)
